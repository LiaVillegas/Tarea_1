#include <iostream>
#include <string>

int main()
{
	int e, n0, n1, n2, n3, n4;
	
	std::cout << "Ingrese un número entero de 5 dígitos: " << std::endl;
	std::cin >> e;
	
	n0 = e / 10000;
	n1 = e % 10000 / 1000;
	n2 = e % 1000 / 100;
	n3 = e % 100 / 10;
	n4 = e % 10;
	
	std::cout << n0 << "   " << n1 << "   "<< n2 << "   " << n3 << "   " << n4 << std::endl;
	
	return 0; 
	
}
